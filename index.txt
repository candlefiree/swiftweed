// 等待页面所有DOM元素加载完成后再执行，避免找不到元素报错
document.addEventListener('DOMContentLoaded', function () {
    // ========== 模态框功能（原代码完整保留） ==========
    function openModal(threadId) {
        const threadDataElement = document.getElementById(threadId);
        if (!threadDataElement) return;

        const title = threadDataElement.getAttribute('data-title');
        const htmlContent = threadDataElement.innerHTML;

        document.getElementById('modalTitle').innerText = title;
        document.getElementById('modalContent').innerHTML = htmlContent;
        document.getElementById('modalOverlay').style.display = 'block';
    }

    function closeModal() {
        document.getElementById('modalOverlay').style.display = 'none';
        document.getElementById('modalContent').innerHTML = '';
    }

    document.getElementById('modalOverlay').addEventListener('click', function (e) {
        if (e.target === this) closeModal();
    });

    // ========== 音乐播放功能（容错优化版） ==========
    let isPlaying = false;
    const bgMusic = document.getElementById('bgMusic');
    const musicText = document.getElementById('musicText');

    // 容错：如果页面没找到音乐元素，直接跳过不执行后续逻辑
    if (!bgMusic || !musicText) return;

    function toggleMusic() {
        if (isPlaying) {
            bgMusic.pause();
            musicText.innerText = "PLAY BGM";
            isPlaying = false;
        } else {
            bgMusic.play().then(() => {
                musicText.innerText = "PAUSE";
                isPlaying = true;
            }).catch(err => {
                alert("音乐加载失败，请检查MP3文件路径或链接是否有效");
            });
        }
    }

    // 将函数挂载到全局，确保HTML里的onclick能正常调用
    window.openModal = openModal;
    window.closeModal = closeModal;
    window.toggleMusic = toggleMusic;
});
